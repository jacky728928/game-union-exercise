JavaScript game.js - 核心游戏类 
class Game {
    constructor() {
        this.state = 'menu';
        this.gold = 10;
        this.round = 1;
        this.campLevel = 1;
        this.heroHP = 30;
        this.playerUnits = [];
        this.enemyUnits = [];
    }

    initGame() {
        this.state = 'playing';
        this.setupBoard();
        this.startRound();
    }

    startRound() {
        this.refreshShop();
        this.startBattle();
    }

    upgradeCamp() {
        if (this.gold >= this.getUpgradeCost()) {
            this.gold -= this.getUpgradeCost();
            this.campLevel++;
            this.updateUI();
        }
    }
}
JavaScript battle.js - 回合制战斗 
class BattleSystem {
    constructor(game) {
        this.game = game;
        this.battleSpeed = 1;
    }

    async executeRound() {
        const allUnits = [...this.game.playerUnits, 
                         ...this.game.enemyUnits];
        
        for (const unit of allUnits) {
            if (unit.isAlive()) {
                await this.processUnitTurn(unit);
                await this.delay(500 / this.battleSpeed);
            }
        }
        
        this.checkBattleEnd();
    }

    processUnitTurn(unit) {
        const target = this.findTarget(unit);
        if (target) {
            const damage = this.calculateDamage(unit, target);
            target.takeDamage(damage);
            this.showDamageEffect(unit, target, damage);
        }
    }
}
JavaScript chess.js - 棋子与技能 
class ChessUnit {
    constructor(config) {
        this.id = config.id;
        this.name = config.name;
        this.hp = config.hp;
        this.maxHp = config.maxHp;
        this.atk = config.atk;
        this.stars = config.stars || 1;
        this.skill = new Skill(config.skillConfig);
    }

    takeDamage(amount) {
        this.hp = Math.max(0, this.hp - amount);
        if (this.hp <= 0) {
            this.onDeath();
        }
    }

    useSkill(targets) {
        return this.skill.execute(this, targets);
    }

    canMerge(other) {
        return this.name === other.name && 
               this.stars === other.stars;
    }
}
