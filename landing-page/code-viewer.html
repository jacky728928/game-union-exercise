class Game {
    constructor() {
        // 游戏状态
        this.state = 'menu';
        this.gold = 10;
        this.round = 1;
        this.campLevel = 1;
        this.heroHP = 30;
        
        // 棋子数据
        this.playerUnits = [];
        this.enemyUnits = [];
        this.benchUnits = [];
        this.shopItems = [];
        
        // UI引用
        this.ui = new GameUI();
        
        this.initEventListeners();
    }

    initEventListeners() {
        // 绑定游戏事件
        document.addEventListener('goldChange', 
            this.onGoldChange.bind(this));
        document.addEventListener('roundChange', 
            this.onRoundChange.bind(this));
    }

    initGame() {
        this.state = 'playing';
        this.setupBoard();
        this.refreshShop();
        this.ui.updateUI();
    }

    startRound() {
        this.round++;
        this.gold += this.getIncome();
        
        // AI生成敌方棋子
        this.generateEnemyUnits();
        
        this.ui.updateUI();
    }

    upgradeCamp() {
        const cost = this.getUpgradeCost();
        if (this.gold >= cost) {
            this.gold -= cost;
            this.campLevel++;
            this.ui.showUpgradeEffect();
        }
    }

    getUpgradeCost() {
        return this.campLevel * 10;
    }

    getIncome() {
        return 3 + this.campLevel;
    }
}

class BattleSystem {
    constructor(game) {
        this.game = game;
        this.battleSpeed = 1;
        this.isBattleActive = false;
        this.battleLog = [];
    }

    async startBattle() {
        this.isBattleActive = true;
        this.showBattleScene();
        
        while (this.isBattleActive) {
            await this.executeBattleRound();
            if (this.checkBattleEnd()) {
                this.isBattleActive = false;
            }
        }
        
        this.endBattle();
    }

    async executeBattleRound() {
        // 获取所有存活单位并按速度排序
        const units = [...this.game.playerUnits, 
                          ...this.game.enemyUnits]
            .filter(u => u.hp > 0)
            .sort((a, b) => b.speed - a.speed);

        for (const unit of units) {
            if (!this.isBattleActive) break;
            
            await this.processUnitTurn(unit);
            await this.delay(600 / this.battleSpeed);
        }
    }

    processUnitTurn(unit) {
        // 优先释放技能
        if (unit.canUseSkill()) {
            this.executeSkill(unit);
        } else {
            // 普通攻击
            const target = this.findTarget(unit);
            if (target) {
                this.executeAttack(unit, target);
            }
        }
    }

    calculateDamage(attacker, defender) {
        let damage = attacker.atk;
        
        // 护甲减伤计算
        damage *= (100 / (100 + defender.def));
        
        // 暴击判定
        if (Math.random() < attacker.critRate) {
            damage *= 1.5;
        }
        
        return Math.floor(damage);
    }

    delay(ms) {
        return new Promise(resolve => {
            setTimeout(resolve, ms);
        });
    }
}

class ChessUnit {
    constructor(config) {
        this.id = config.id;
        this.name = config.name;
        this.element = config.element;
        this.rarity = config.rarity;
        
        // 基础属性
        this.baseHp = config.hp;
        this.baseAtk = config.atk;
        this.baseDef = config.def;
        this.speed = config.speed || 100;
        
        // 当前属性（受星级影响）
        this.stars = config.stars || 1;
        this.recalculateStats();
        
        // 技能系统
        this.skill = new Skill(config.skill);
        this.skillCooldown = 0;
    }

    recalculateStats() {
        // 星级提升属性倍率
        const multiplier = 1 + (this.stars - 1) * 0.5;
        this.hp = Math.floor(this.baseHp * multiplier);
        this.maxHp = this.hp;
        this.atk = Math.floor(this.baseAtk * multiplier);
        this.def = Math.floor(this.baseDef * multiplier);
    }

    takeDamage(amount) {
        this.hp = Math.max(0, this.hp - amount);
        
        if (this.hp <= 0) {
            this.onDeath();
        }
    }

    canUseSkill() {
        return this.skillCooldown <= 0 && 
               this.hp < this.maxHp * 0.7;
    }

    useSkill(targets) {
        this.skillCooldown = this.skill.cooldown;
        return this.skill.execute(this, targets);
    }

    canMerge(other) {
        return this.name === other.name && 
               this.stars === other.stars && 
               this.stars < 3;
    }

    merge(other) {
        if (this.canMerge(other)) {
            this.stars++;
            this.recalculateStats();
            return true;
        }
        return false;
    }
}

class MenuSystem {
    constructor() {
        this.overlays = {
            settings: document.getElementById('settings-overlay'),
            tutorial: document.getElementById('tutorial-overlay'),
            announcement: document.getElementById('announcement-overlay'),
            gameOver: document.getElementById('game-over')
        };
        
        this.gameContainer = document.getElementById('game-container');
        
        this.bindEvents();
    }

    bindEvents() {
        // 设置界面
        document.getElementById('close-settings')
            .addEventListener('click', this.closeSettings.bind(this));
        
        // 教程界面
        document.getElementById('tutorial-next')
            .addEventListener('click', this.nextTutorial.bind(this));
    }

    showOverlay(overlayName) {
        if (this.overlays[overlayName]) {
            this.overlays[overlayName].classList.remove('hidden');
            document.body.style.overflow = 'hidden';
        }
    }

    hideOverlay(overlayName) {
        if (this.overlays[overlayName]) {
            this.overlays[overlayName].classList.add('hidden');
            document.body.style.overflow = 'auto';
        }
    }

    showGame() {
        this.gameContainer.classList.remove('hidden');
        document.getElementById('start-menu').classList.add('hidden');
    }

    showStartMenu() {
        this.gameContainer.classList.add('hidden');
        document.getElementById('start-menu').classList.remove('hidden');
    }

    updateStats(stats) {
        if (stats.gold !== undefined) {
            document.getElementById('gold').textContent = stats.gold;
        }
        if (stats.round !== undefined) {
            document.getElementById('round').textContent = stats.round;
        }
        if (stats.campLevel !== undefined) {
            document.getElementById('camp-level').textContent = stats.campLevel;
        }
    }
}
